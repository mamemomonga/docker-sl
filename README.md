	$ docker run -t --rm mamemomonga/sl
	$ docker run -t --rm mamemomonga/sl -l

	$ alias sl="docker run -t --rm mamemomonga/sl"
	$ sl
	$ sl -a
	$ sl -l
	$ sl -F
	$ sl -c
